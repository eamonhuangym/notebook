# 栈、队列、列表

## 队列

特点：先进先出

输入9个数，输出其队列

~~~c
#include <stdio.h>
struct queue{
	int data[100];
	int head;
	int tail;
};
int main()
{
	struct queue q;
	//下面初始化队列 
	q.head = 1;
	q.tail = 1;
	for(int i=1;i<=9;i++){
		scanf("%d",&q.data[q.tail]);
		q.tail++;
	}
	while(q.head<q.tail){
		printf("%d ",q.data[q.head]);
		q.head++;
		
		q.data[q.tail]=q.data[q.head];
		q.tail++;
		q.head++;
	}
	return 0;
}
~~~
